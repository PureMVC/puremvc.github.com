     6  import org.puremvc.as2.interfaces.*;
     7  
     8  /**
     9   * A base <code>INotification</code> implementation.
    10   * 
    11   * <P>
    12   * PureMVC does not rely upon underlying event models such 
    13   * as the one provided with Flash, and ActionScript 3 does 
    14   * not have an inherent event model.</P>
    15   * 
    16   * <P>
    17   * The Observer Pattern as implemented within PureMVC exists 
    18   * to support event-driven communication between the 
    19   * application and the actors of the MVC triad.</P>
    20   * 
    21   * <P>
    22   * Notifications are not meant to be a replacement for Events
    23   * in Flex/Flash/AIR. Generally, <code>IMediator</code> implementors
    24   * place event listeners on their view components, which they
    25   * then handle in the usual way. This may lead to the broadcast of <code>Notification</code>s to 
    26   * trigger <code>ICommand</code>s or to communicate with other <code>IMediators</code>. <code>IProxy</code> and <code>ICommand</code>
    27   * instances communicate with each other and <code>IMediator</code>s 
    28   * by broadcasting <code>INotification</code>s.</P>
    29   * 
    30   * <P>
    31   * A key difference between Flash <code>Event</code>s and PureMVC 
    32   * <code>Notification</code>s is that <code>Event</code>s follow the 
    33   * 'Chain of Responsibility' pattern, 'bubbling' up the display hierarchy 
    34   * until some parent component handles the <code>Event</code>, while
    35   * PureMVC <code>Notification</code>s follow a 'Publish/Subscribe'
    36   * pattern. PureMVC classes need not be related to each other in a 
    37   * parent/child relationship in order to communicate with one another
    38   * using <code>Notification</code>s.
    39   * 
    40   * @see org.puremvc.as2.patterns.observer.Observer Observer
    41   * 
    42   */
    43  class org.puremvc.as2.patterns.observer.Notification implements INotification
    44  {
    45  	
    46  	/**
    47  	 * Constructor. 
    48  	 * 
    49  	 * @param name name of the <code>Notification</code> instance. (required)
    50  	 * @param body the <code>Notification</code> body. (optional)
    51  	 * @param type the type of the <code>Notification</code> (optional)
    52  	 */
    53  	public function Notification( name:String, body:Object, type:String )
    54  	{
    55  		this.name = name;
    56  		this.body = body;
    57  		this.type = type;
    58  	}
    59  	
    60  	/**
    61  	 * Get the name of the <code>Notification</code> instance.
    62  	 * 
    63  	 * @return the name of the <code>Notification</code> instance.
    64  	 */
    65  	public function getName() : String
    66  	{
    67  		return name;
    68  	}
    69  	
    70  	/**
    71  	 * Set the body of the <code>Notification</code> instance.
    72  	 */
    73  	public function setBody( body:Object ) : Void
    74  	{
    75  		this.body = body;
    76  	}
    77  	
    78  	/**
    79  	 * Get the body of the <code>Notification</code> instance.
    80  	 * 
    81  	 * @return the body object. 
    82  	 */
    83  	public function getBody() : Object
    84  	{
    85  		return body;
    86  	}
    87  	
    88  	/**
    89  	 * Set the type of the <code>Notification</code> instance.
    90  	 */
    91  	public function setType( type:String ) : Void
    92  	{
    93  		this.type = type;
    94  	}
    95  	
    96  	/**
    97  	 * Get the type of the <code>Notification</code> instance.
    98  	 * 
    99  	 * @return the type  
   100  	 */
   101  	public function getType() : String
   102  	{
   103  		return type;
   104  	}
   105  
   106  	/**
   107  	 * Get the string representation of the <code>Notification</code> instance.
   108  	 * 
   109  	 * @return the string representation of the <code>Notification</code> instance.
   110  	 */
   111  	public function toString() : String
   112  	{
   113  		var msg : String = "Notification Name: "+getName();
   114  		msg += "\nBody:"+( body == null )?"null":body.toString();
   115  		msg += "\nType:"+( type == null )?"null":type;
   116  		return msg;
   117  	}
   118  	
   119  	// the name of the notification instance
   120  	private var name	: String;
   121  	// the type of the notification instance
   122  	private var type : String;
   123  	// the body of the notification instance
   124  	private var body : Object;
   125  		
   126  }
