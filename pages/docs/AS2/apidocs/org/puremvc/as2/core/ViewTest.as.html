    55  	/**
    56  	 * Tests registration and notification of Observers.
    57  	 * 
    58  	 * An Observer is created to callback the viewTestMethod of
    59  	 * this ViewTest instance. This Observer is registered with
    60  	 * the View to be notified of 'ViewTestEvent' events. Such
    61  	 * an event is created, and a value set on its payload. Then
    62  	 * the View is told to notify interested observers of this
    63  	 * Event. 
    64  	 * 
    65  	 * The View calls the Observer's notifyObserver method
    66  	 * which calls the viewTestMethod on this instance
    67  	 * of the ViewTest class. The viewTestMethod method will set 
    68  	 * an instance variable to the value passed in on the Event
    69  	 * payload. We evaluate the instance variable to be sure
    70  	 * it is the same as that passed out as the payload of the 
    71  	 * original 'ViewTestEvent'.
    72  	 * 
    73  	 */
    74  	public function testRegisterAndNotifyObserver():Void {
    75  		
    76  		// Get the Singleton View instance
    77  		var view:IView = View.getInstance();
    78  		
    79  		// Create observer, passing in notification method and context
    80  		var observer:Observer = new Observer(viewTestMethod, this);
    81  		
    82  		// Register Observer's interest in a particular Notification with the View 
    83  		view.registerObserver(MockNotification.NAME, observer);
    84  		
    85  		// Create a MockNotification, setting 
    86  		// a body value, and tell the View to notify 
    87  		// Observers. Since the Observer is this class 
    88  		// and the notification method is viewTestMethod,
    89  		// successful notification will result in our local 
    90  		// viewTestVar being set to the value we pass in 
    91  		// on the note body.
    92  		var note:INotification = MockNotification.create(10);
    93  		view.notifyObservers(note);
    94  	}

    96  	/**
    97  	 * A utility method to test the notification of Observers by the view
    98  	 */
    99  	private function viewTestMethod( note:INotification ):Void {
   100  		// set the local viewTestVar to the number on the event payload
   101  		// test assertions  			
   102  		assertTrue("Expecting viewTestVar = 10", Number(note.getBody()) == 10);
   103  	}

   233  	/**
   234  	 * Tests registering a Mediator for 2 different notifications, removing the
   235  	 * Mediator from the View, and seeing that neither notification causes the
   236  	 * Mediator to be notified.
   237  	 */
   238  	public function testRemoveMediatorAndSubsequentNotify():Void {
   239  		
   240  		// Get the Singleton View instance
   241  		var view:IView = View.getInstance();
   242  		
   243  		// Create and register the test mediator to be removed.
   244  		view.registerMediator(new MockMediator(this, [ViewTest.NOTE1, ViewTest.NOTE2]));
   245  		
   246  		// test that notifications work
   247  		view.notifyObservers(new Notification(NOTE1));
   248  		assertTrue("Expecting lastNotification == NOTE1", lastNotification == NOTE1);
   249  
   250  		view.notifyObservers(new Notification(NOTE2));
   251  		assertTrue("Expecting lastNotification == NOTE2", lastNotification == NOTE2);
   252  
   253  		// Remove the Mediator
   254  		view.removeMediator(MockMediator.NAME);
   255  
   256  		// test that retrieving it now returns null			
   257  		assertTrue("Expecting view.retrieveMediator( MockMediator.NAME ) == null", view.retrieveMediator(MockMediator.NAME) == null);
   258  
   259  		// test that notifications no longer work
   260  		lastNotification = null;
   261  		
   262  		view.notifyObservers(new Notification(NOTE1));
   263  		assertTrue("Expecting lastNotification != NOTE1", lastNotification != NOTE1);
   264  
   265  		view.notifyObservers(new Notification(NOTE2));
   266  		assertTrue("Expecting lastNotification != NOTE2", lastNotification != NOTE2);
   267  	}

   270  	 * Tests registering one of two registered Mediators and seeing
   271  	 * that the remaining one still responds.
   272  	 */
   273  	public function testRemoveOneOfTwoMediatorsAndSubsequentNotify():Void {
   274  		
   275  		// Get the Singleton View instance
   276  		var view:IView = View.getInstance();
   277  		
   278  		// Create and register that responds to notifications 1 and 2
   279  		view.registerMediator(new MockMediator(this, [ViewTest.NOTE1, ViewTest.NOTE2]));
   280  		
   281  		// Create and register that responds to notification 3
   282  		view.registerMediator(new MockMediator(this, [ViewTest.NOTE3], "note3Mediator"));
   283  		
   284  		// test that all notifications work
   285  		view.notifyObservers(new Notification(NOTE1));
   286  		assertTrue("Expecting lastNotification == NOTE1", lastNotification == NOTE1);
   287  
   288  		view.notifyObservers(new Notification(NOTE2));
   289  		assertTrue("Expecting lastNotification == NOTE2", lastNotification == NOTE2);
   290  
   291  		view.notifyObservers(new Notification(NOTE3));
   292  		assertTrue("Expecting lastNotification == NOTE3", lastNotification == NOTE3);
   293  				
   294  		// Remove the Mediator that responds to 1 and 2
   295  		view.removeMediator(MockMediator.NAME);
   296  
   297  		// test that retrieving it now returns null			
   298  		assertTrue("Expecting view.retrieveMediator( MockMediator.NAME ) == null", view.retrieveMediator(MockMediator.NAME) == null);
   299  
   300  		// test that notifications no longer work
   301  		// for notifications 1 and 2, but still work for 3
   302  		lastNotification = null;
   303  		
   304  		view.notifyObservers(new Notification(NOTE1));
   305  		assertTrue("Expecting lastNotification != NOTE1", lastNotification != NOTE1);
   306  
   307  		view.notifyObservers(new Notification(NOTE2));
   308  		assertTrue("Expecting lastNotification != NOTE2", lastNotification != NOTE2);
   309  
   310  		view.notifyObservers(new Notification(NOTE3));
   311  		assertTrue("Expecting lastNotification == NOTE3", lastNotification == NOTE3);
   312  	}
