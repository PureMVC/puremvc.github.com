     7  /**
     8   * The interface definition for a PureMVC Notifier.
     9   * 
    10   * <P>
    11   * <code>MacroCommand, Command, Mediator</code> and <code>Proxy</code>
    12   * all have a need to send <code>Notifications</code>. </P>
    13   * 
    14   * <P>
    15   * The <code>INotifier</code> interface provides a common method called
    16   * <code>sendNotification</code> that relieves implementation code of 
    17   * the necessity to actually construct <code>Notifications</code>.</P>
    18   * 
    19   * <P>
    20   * The <code>Notifier</code> class, which all of the above mentioned classes
    21   * extend, also provides an initialized reference to the <code>Facade</code>
    22   * Singleton, which is required for the convienience method
    23   * for sending <code>Notifications</code>, but also eases implementation as these
    24   * classes have frequent <code>Facade</code> interactions and usually require
    25   * access to the facade anyway.</P>
    26   * 
    27   * @see org.puremvc.as2.interfaces.IFacade IFacade
    28   * @see org.puremvc.as2.interfaces.INotification INotification
    29   */
    30  interface org.puremvc.as2.interfaces.INotifier
    31  {
    32  	
    33  	/**
    34  	 * Send a <code>INotification</code>.
    35  	 * 
    36  	 * <P>
    37  	 * Convenience method to prevent having to construct new 
    38  	 * notification instances in our implementation code.</P>
    39  	 * 
    40  	 * @param notificationName the name of the notification to send
    41  	 * @param body the body of the notification (optional)
    42  	 * @param type the type of the notification (optional)
    43  	 */ 
    44  	function sendNotification( notificationName:String, body:Object, type:String ):Void; 
    45  		
    46  }
