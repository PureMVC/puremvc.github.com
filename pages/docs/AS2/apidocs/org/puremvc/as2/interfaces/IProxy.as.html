     7  /**
     8   * The interface definition for a PureMVC Proxy.
     9   *
    10   * <P>
    11   * In PureMVC, <code>IProxy</code> implementors assume these responsibilities:</P>
    12   * <UL>
    13   * <LI>Implement a common method which returns the name of the Proxy.</LI>
    14   * </UL>
    15   * <P>
    16   * Additionally, <code>IProxy</code>s typically:</P>
    17   * <UL>
    18   * <LI>Maintain references to one or more pieces of model data.</LI>
    19   * <LI>Provide methods for manipulating that data.</LI>
    20   * <LI>Generate <code>INotifications</code> when their model data changes.</LI>
    21   * <LI>Expose their name as a <code>public static const</code> called <code>NAME</code>.</LI>
    22   * <LI>Encapsulate interaction with local or remote services used to fetch and persist model data.</LI>
    23   * </UL>
    24   */
    25  interface org.puremvc.as2.interfaces.IProxy
    26  {
    27  
    28  	/**
    29  	 * Get the Proxy name
    30  	 * 
    31  	 * @return the Proxy instance name
    32  	 */
    33  	function getProxyName():String;
    34  	
    35  	/**
    36  	 * Set the data object
    37  	 * 
    38  	 * @param data the data object
    39  	 */
    40  	function setData( data:Object ):Void;
    41  	
    42  	/**
    43  	 * Get the data object
    44  	 * 
    45  	 * @return the data as type Object
    46  	 */
    47  	function getData():Object; 
    48          
    49      /**
    50       * Called by the Model when the Proxy is registered
    51       */ 
    52      function onRegister( ):Void;
    53  
    54      /**
    55       * Called by the Model when the Proxy is removed
    56       */ 
    57      function onRemove( ):Void;
    58  
    59  }
