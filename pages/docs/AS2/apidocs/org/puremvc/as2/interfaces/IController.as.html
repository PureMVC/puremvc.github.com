     6  import org.puremvc.as2.interfaces.*;
     7  
     8  /**
     9   * The interface definition for a PureMVC Controller.
    10   * 
    11   * <P>
    12   * In PureMVC, an <code>IController</code> implementor 
    13   * follows the 'Command and Controller' strategy, and 
    14   * assumes these responsibilities:
    15   * <UL>
    16   * <LI> Remembering which <code>ICommand</code>s 
    17   * are intended to handle which <code>INotifications</code>.</LI>
    18   * <LI> Registering itself as an <code>IObserver</code> with
    19   * the <code>View</code> for each <code>INotification</code> 
    20   * that it has an <code>ICommand</code> mapping for.</LI>
    21   * <LI> Locating the proper <code>ICommand</code> instance
    22   * to handle a given <code>INotification</code> when notified by the <code>View</code>.</LI>
    23   * <LI> Calling the <code>ICommand</code>'s <code>execute</code>
    24   * method, passing in the <code>INotification</code>.</LI> 
    25   * </UL>
    26   *
    27   * @see org.puremvc.as2.interfaces INotification
    28   * @see org.puremvc.as2.interfaces ICommand
    29   */
    30  interface org.puremvc.as2.interfaces.IController
    31  {
    32  	
    33  	/**
    34  	 * Register a particular <code>ICommand</code> class as the handler 
    35  	 * for a particular <code>INotification</code>.
    36  	 * 
    37  	 * @param notificationName the name of the <code>INotification</code>
    38  	 * @param commandClassRef the Class of the <code>ICommand</code>
    39  	 *
    40  	 */
    41  	function registerCommand( notificationName:String, commandClassRef:Function ) : Void;
    42  	
    43  	/**
    44  	 * Execute the <code>ICommand</code> previously registered as the
    45  	 * handler for <code>INotification</code>s with the given notification name.
    46  	 * 
    47  	 * @param notification the <code>INotification</code> to execute the associated <code>ICommand</code> for
    48  	 */
    49  	function executeCommand( notification : INotification ) : Void;
    50  
    51  	/**
    52  	 * Remove a previously registered <code>ICommand</code> to <code>INotification</code> mapping.
    53  	 * 
    54  	 * @param notificationName the name of the <code>INotification</code> to remove the <code>ICommand</code> mapping for
    55  	 */
    56  	function removeCommand( notificationName : String ):Void;
    57  	
    58  	/**
    59  	 * Check if a Command is registered for a given Notification 
    60  	 * 
    61  	 * @param notificationName
    62  	 * @return whether a Command is currently registered for the given <code>notificationName</code>.
    63  	 */
    64  	function hasCommand( notificationName:String ) : Boolean;
    65  	
    66  }
