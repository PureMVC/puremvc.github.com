     6  import org.puremvc.as2.interfaces.*;
     7  import org.puremvc.as2.patterns.command.*;
     8  import org.puremvc.as2.patterns.proxy.MockVO;
     9  
    10  /**
    11   * A SimpleCommand subclass used by ControllerTest.
    12   * Execute defaults to changing the 'result' property of the VO, multiplying the 
    13   * input by 2
    14   * @see org.puremvc.as2.core.controller.ControllerTest ControllerTest
    15   * @see org.puremvc.as2.core.controller.MockVO MockVO
    16   */
    17  class org.puremvc.as2.patterns.command.MockSimpleCommand extends SimpleCommand implements ICommand {
    18  
    19  	private var _multiplier:Number;
    20  	private var _receiver:String; //String name of property in receiving VO
    21  	
    22  	/**
    23  	 * Constructor.
    24  	 */
    25  	public function MockSimpleCommand(multiplier:Number, receiver:String) {
    26  		super();
    27  		_multiplier = multiplier || 2;
    28  		_receiver = receiver || "result";
    29  	}
    30  	
    31  	/**
    32  	 * Fabricate a result by multiplying the input by 2
    33  	 * 
    34  	 * @param note the note carrying the ControllerTestVO
    35  	 */
    36  	public function execute( note:INotification ):Void {
    37  		
    38  		var vo:MockVO = MockVO(note.getBody());
    39  		// Fabricate a result
    40  		vo[_receiver]= _multiplier * vo.input;
    41  	}
    42  	
    43  }
