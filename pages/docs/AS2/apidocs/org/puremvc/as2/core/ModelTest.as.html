     6  import org.puremvc.as2.core.Model;

    10  import org.puremvc.as2.patterns.proxy.Proxy;

    12  class org.puremvc.as2.core.ModelTest extends com.asunit.framework.TestCase {
    13  
    14  	private var className:String = "org.puremvc.as2.core.ModelTest";
    15  	private var instance:IModel;
    16  
    17  	public function setUp():Void {
    18  		instance = Model.getInstance();
    19  	}
    20  
    21  	public function tearDown():Void {
    22  		delete instance;
    23   	}

    26  	/**
    27  	* Tests the Model Singleton Factory Method 
    28  	*/
    29  	public function testGetInstance():Void {
    30  		assertNotNull("Expecting instance not null", instance);
    31  		assertTrue("Expecting instance implements IModel", instance instanceof IModel);
    32  		
    33  		var model:IModel = Model.getInstance();
    34  		assertSame("2 Model variable point to the same Model in memory", instance, model);
    35  	}	
    36  	
    37  	/**
    38  	 * Tests the proxy registration and retrieval methods.
    39  	 * 
    40  	 * Tests {@link Model#registerProxy} and {@link Model#retrieveProxy} in the same test.
    41  	 * 
    42  	 * These methods cannot currently be tested separately
    43  	 * in any meaningful way other than to show that the
    44  	 * methods do not throw exceptions when called.
    45  	 */
    46  	public function testRegisterAndRetrieveProxy():Void{
    47  		
    48  		// register a proxy and retrieve it.
    49  		var model:IModel = Model.getInstance();
    50  		model.registerProxy(new Proxy("colors", ["red", "green", "blue"]));
    51  		var proxy:Proxy = Proxy(model.retrieveProxy("colors"));
    52  		var data = proxy.getData();
    53  		
    54  		// test assertions
    55  		assertNotNull( "Expecting data not null", data);
    56  		assertTrue( "Expecting data type is Array", data instanceof Array);
    57  		assertEquals( "Expecting data.length == 3", 3, data.length);
    58  		assertEquals( "Expecting data[0] == 'red'", "red", data[0]);
    59  		assertEquals( "Expecting data[1] == 'green'", "green", data[1]);
    60  		assertEquals( "Expecting data[2] == 'blue'", "blue", data[2]);
    61  	}
    62  	
    63  	/**
    64  	 * Tests the proxy removal method.
    65  	 */
    66  	public function testRegisterAndRemoveProxy():Void{
    67  		
    68  		// register a proxy, remove it, then try to retrieve it
    69  		var model:IModel = Model.getInstance();
    70  		var proxy:IProxy = new Proxy("sizes", ["7", "13", "21"]);
    71  		model.registerProxy(proxy);
    72  
    73  		assertNotNull("proxy is registered and retrievable", model.retrieveProxy("sizes"));
    74  
    75  		// remove the proxy
    76  		var removedProxy:IProxy = model.removeProxy("sizes");
    77  		
    78  		// assert that we removed the appropriate proxy
    79  		assertTrue("Expecting removedProxy.getProxyName() == 'sizes'", 
    80  					removedProxy.getProxyName() == "sizes");
    81  		
    82  		// ensure that the proxy is no longer retrievable from the model
    83  		assertFalse("model.hasProxy('sizes') == false", model.hasProxy("sizes"));
    84  		proxy = model.retrieveProxy("sizes");
    85  		assertNull( "Expecting proxy is null", proxy );
    86  	}
    87  	
    88  	/**
    89  	 * Tests the hasProxy Method
    90  	 */
    91  	public function testHasProxy():Void{
    92  		
    93  		// register a proxy
    94  		var model:IModel = Model.getInstance();
    95  		var proxy:IProxy = new Proxy("aces", ["clubs", "spades", "hearts", "diamonds"]);
    96  		model.registerProxy(proxy);
    97  		
    98  		// assert that the model.hasProxy method returns true
    99  		// for that proxy name
   100  		assertTrue("Expecting model.hasProxy('aces') == true", 
   101  					model.hasProxy("aces") == true);
   102  		
   103  		// remove the proxy
   104  		model.removeProxy("aces");
   105  		
   106  		// assert that the model.hasProxy method returns false
   107  		// for that proxy name
   108  		assertTrue("Expecting model.hasProxy('aces') == false", 
   109  					model.hasProxy("aces") == false);
   110  	}
   111  	
   112  	/**
   113  	 * Tests that the Model calls the onRegister and onRemove methods
   114  	 */
   115  	public function testOnRegisterAndOnRemove():Void{
   116  		
   117  		// Get the Singleton View instance
   118  		var model:IModel = Model.getInstance();
   119  
   120  		// Create and register the test proxy
   121  		var proxy:IProxy = new MockProxy();
   122  		// check that there's no data before registration
   123  		assertTrue("Before registering with model, proxy.getData() == ''", 
   124  					proxy.getData() == "");
   125  		
   126  		model.registerProxy( proxy);
   127  
   128  		// assert that onRegister was called, and the proxy responded by setting its data accordingly
   129  		assertTrue("Expecting proxy.getData() == MockProxy.ON_REGISTER_CALLED", 
   130  					proxy.getData() == MockProxy.ON_REGISTER_CALLED);
   131  		
   132  		// Remove the component
   133  		model.removeProxy(MockProxy.NAME);
   134  
   135  		// assert that onRemove was called, and the proxy responded by setting its data accordingly
   136  		assertTrue("Expecting proxy.getData() == MockProxy.ON_REMOVE_CALLED", 
   137  					proxy.getData() == MockProxy.ON_REMOVE_CALLED);
   138  	}
