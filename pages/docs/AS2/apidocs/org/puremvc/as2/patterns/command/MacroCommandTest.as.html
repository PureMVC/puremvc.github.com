     6  import org.puremvc.as2.interfaces.*;

    12  class org.puremvc.as2.patterns.command.MacroCommandTest extends com.asunit.framework.TestCase {
    13  	
    14  	private var className:String = "org.puremvc.as2.patterns.command.MacroCommandTest";
    15  	private var instance:MacroCommand;
    16  
    17  	public function setUp():Void {
    18  		instance = new MacroCommand();
    19  	}
    20  
    21  	public function tearDown():Void {
    22  		delete instance;
    23   	}
    24  
    25   	public function testInstantiated():Void {
    26  		assertTrue("MacroCommand instantiated", instance instanceof MacroCommand);
    27  	}
    28  	
    29  /**
    30   * Tests operation of a <code>MacroCommand</code>.
    31  	 * 
    32  	 * <P>
    33  	 * This test creates a new <code>Notification</code>, adding a 
    34  	 * <code>MacroCommandTestVO</code> as the body. 
    35  	 * It then creates a <code>MockMacroCommand</code> and invokes
    36  	 * its <code>execute</code> method, passing in the 
    37  	 * <code>Notification</code>.<P>
    38  	 * 
    39  	 * <P>
    40  	 * The <code>MockMacroCommand</code> has defined an
    41  	 * <code>initializeMacroCommand</code> method, which is 
    42  	 * called automatically by its constructor. In this method
    43  	 * the <code>MockMacroCommand</code> adds 2 SubCommands
    44  	 * to itself, <code>MacroCommandTestSub1Command</code> and
    45  	 * <code>MacroCommandTestSub2Command</code>.
    46  	 * 
    47  	 * <P>
    48  	 * The <code>MacroCommandTestVO</code> has 2 result properties,
    49  	 * one is set by <code>MacroCommandTestSub1Command</code> by
    50  	 * multiplying the input property by 2, and the other is set
    51  	 * by <code>MacroCommandTestSub2Command</code> by multiplying
    52  	 * the input property by itself. 
    53  	 * 
    54  	 * <P>
    55  	 * Success is determined by evaluating the 2 result properties
    56  	 * on the <code>MacroCommandTestVO</code> that was passed to 
    57  	 * the <code>MockMacroCommand</code> on the Notification 
    58  	 * body.</P>
    59  	 * 
    60  	 */
    61  	public function testMacroCommandExecute():Void {
    62  		
    63  		// Create the VO
    64  		var vo:MockVO = new MockVO(5);
    65  		
    66  		// Create the Notification (note)
    67  		var note:INotification = new Notification("MacroCommandTest", vo);
    68  
    69  		// Create the SimpleCommand  			
    70  		var command:MockMacroCommand = new MockMacroCommand();
    71  		
    72  		// Execute the SimpleCommand
    73  		command.execute(note);
    74  		
    75  		// test assertions
    76  		assertTrue( "Expecting vo.result1 == 10", vo.result1 == 10 );
    77  		assertTrue( "Expecting vo.result2 == 25", vo.result2 == 25 );
    78  		
    79  	}
    80  
    81  }
