     6  import org.puremvc.as2.interfaces.*;
     7  
     8  /**
     9   * The interface definition for a PureMVC Facade.
    10   *
    11   * <P>
    12   * The Facade Pattern suggests providing a single
    13   * class to act as a central point of communication 
    14   * for a subsystem. </P>
    15   * 
    16   * <P>
    17   * In PureMVC, the Facade acts as an interface between
    18   * the core MVC actors (Model, View, Controller) and
    19   * the rest of your application.</P>
    20   * 
    21   * @see org.puremvc.as2.interfaces.IModel IModel
    22   * @see org.puremvc.as2.interfaces.IView IView
    23   * @see org.puremvc.as2.interfaces.IController IController
    24   * @see org.puremvc.as2.interfaces.ICommand ICommand
    25   * @see org.puremvc.as2.interfaces.INotification INotification
    26   */
    27  interface org.puremvc.as2.interfaces.IFacade extends INotifier
    28  {
    29  
    30  	/**
    31  	 * Notify <code>Observer</code>s of an <code>INotification</code>.
    32  	 * 
    33  	 * @param note the <code>INotification</code> to have the <code>View</code> notify observers of.
    34  	 */
    35  	function notifyObservers( note:INotification ) : Void;
    36  	
    37  	/**
    38  	 * Register an <code>IProxy</code> with the <code>Model</code> by name.
    39  	 * 
    40  	 * @param proxy the <code>IProxy</code> to be registered with the <code>Model</code>.
    41  	 */
    42  	function registerProxy( proxy:IProxy ) : Void;
    43  
    44  	/**
    45  	 * Retrieve a <code>IProxy</code> from the <code>Model</code> by name.
    46  	 * 
    47  	 * @param proxyName the name of the <code>IProxy</code> instance to be retrieved.
    48  	 * @return the <code>IProxy</code> previously regisetered by <code>proxyName</code> with the <code>Model</code>.
    49  	 */
    50  	function retrieveProxy( proxyName:String ) : IProxy;
    51  	
    52  	/**
    53  	 * Remove an <code>IProxy</code> instance from the <code>Model</code> by name.
    54  	 *
    55  	 * @param proxyName the <code>IProxy</code> to remove from the <code>Model</code>.
    56  	 */
    57  	function removeProxy( proxyName:String ) : IProxy;
    58  	
    59  	/**
    60  	 * Check if a Proxy is registered
    61  	 * 
    62  	 * @param proxyName
    63  	 * @return whether a Proxy is currently registered with the given <code>proxyName</code>.
    64  	 */
    65  	function hasProxy( proxyName:String ) : Boolean;
    66  		
    67  	/**
    68  	 * Register an <code>ICommand</code> with the <code>Controller</code>.
    69  	 * 
    70  	 * @param noteName the name of the <code>INotification</code> to associate the <code>ICommand</code> with.
    71  	 * @param commandClassRef the Class of the <code>ICommand</code>.
    72  	 */
    73  	function registerCommand( noteName : String, commandClassRef : Function ) : Void
    74  	
    75  	/**
    76  	 * Remove a previously registered <code>ICommand</code> to <code>INotification</code> mapping from the Controller.
    77  	 * 
    78  	 * @param notificationName the name of the <code>INotification</code> to remove the <code>ICommand</code> mapping for
    79  	 */
    80  	function removeCommand( notificationName:String ): Void;
    81  	
    82  	/**
    83  	 * Check if a Command is registered for a given Notification 
    84  	 * 
    85  	 * @param notificationName
    86  	 * @return whether a Command is currently registered for the given <code>notificationName</code>.
    87  	 */
    88  	function hasCommand( notificationName:String ) : Boolean;
    89  		
    90  	/**
    91  	 * Register an <code>IMediator</code> instance with the <code>View</code>.
    92  	 * 
    93  	 * @param mediator a reference to the <code>IMediator</code> instance
    94  	 */
    95  	function registerMediator( mediator:IMediator ) : Void;
    96  
    97  	/**
    98  	 * Retrieve an <code>IMediator</code> instance from the <code>View</code>.
    99  	 * 
   100  	 * @param mediatorName the name of the <code>IMediator</code> instance to retrievve
   101  	 * @return the <code>IMediator</code> previously registered with the given <code>mediatorName</code>.
   102  	 */
   103  	function retrieveMediator( mediatorName:String ) : IMediator;
   104  
   105  	/**
   106  	 * Remove a <code>IMediator</code> instance from the <code>View</code>.
   107  	 * 
   108  	 * @param mediatorName name of the <code>IMediator</code> instance to be removed.
   109  	 */
   110  	function removeMediator( mediatorName:String ) : IMediator;
   111  	
   112  	/**
   113  	 * Check if a Mediator is registered or not
   114  	 * 
   115  	 * @param mediatorName
   116  	 * @return whether a Mediator is registered with the given <code>mediatorName</code>.
   117  	 */
   118  	function hasMediator( mediatorName:String ) : Boolean;
   119  		
   120  }
